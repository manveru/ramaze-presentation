class Calculator < Ramaze::Controller
  def multiply(one, another)
    one.to_i * another.to_i
  end

  def divide(a_number, another_number)
    a_number.to_i / another_number.to_i
  end

  def sum(*numbers)
    numbers.inject(0){|sum, num| sum + num.to_i }
  end
end
